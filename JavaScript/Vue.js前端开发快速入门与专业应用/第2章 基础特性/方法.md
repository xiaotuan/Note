可以通过选项属性 methods 对象来定义方法，并且使用 v-on 指令来监听 DOM 事件。

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="utf-8" />
    <title>Bind Method</title>
    <script src="https://cdn.jsdelivr.net/npm/vue@2.6.11"></script>
  </head>
  <body>
    <button id="app" v-on:click="alert">alert</button>
    <script type="text/javascript">
      var vm = new Vue({
        el: '#app',
        data: { a: 1 },
        methods: {
          alert: function() {
            alert(this.a)
          }
        }
      })
    </script>
  </body>
</html>
```
